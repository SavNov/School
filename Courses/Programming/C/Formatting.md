# Formatting Guide
- %c = char
- %c = unsigned char
- %i, &d = int 
- %u = unsigned int
- %hi, %hd = short int
- %hu = unsigned short int
- %li, %ld = long int
- %lu = unsigned long int
- %lli, %lld = long long int
- %llu = unsigned long long int
- %f = float
*Doubles are just long floats.*
- %lf = double
- %Lf = long double
*Making long doubles just long long floats.*
